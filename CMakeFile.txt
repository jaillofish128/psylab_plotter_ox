project(BloodOxy_draw)

set(CMAKE_AUTOMOC ON)

    about.cpp
	darw.cpp
	histowindow.cpp
	libpsyclass.cpp
	main.cpp
    mainwindow.cpp
    odi.cpp
    qcustomplot.cpp
)

	darw.hpp
	histwondow.h
	libpsyclass.h
	mainwindow.h
	odi.hpp
	qcustomplot.h)
set(UI 
    about.ui
	histowindow.ui
	mainwindow.ui)

add_executable(BloodOxy_draw ${SOURCES} ${HEADERS} ${UI})
qt5_use_modules(BloodOxy_draw Core Network PrintSupport Widgets)
